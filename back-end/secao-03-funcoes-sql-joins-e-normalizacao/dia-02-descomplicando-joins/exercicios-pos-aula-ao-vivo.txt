CREATE TABLE theater (
    id INTEGER auto_increment PRIMARY KEY NOT NULL,
    name VARCHAR(30) NOT NULL,
    location VARCHAR(30) NULL
);
CREATE TABLE movies (
    id INTEGER auto_increment PRIMARY KEY NOT NULL,
    title VARCHAR(30) NOT NULL,
    director VARCHAR(30) NULL,
    year INT NOT NULL,
    length_minutes INT NOT NULL,
    theater_id INTEGER,
    FOREIGN KEY (theater_id) REFERENCES theater (id)
);
CREATE TABLE box_office (
    movie_id INTEGER,
    FOREIGN KEY (movie_id) REFERENCES movies (id),
    rating DECIMAL(2,1) NOT NULL,
    domestic_sales INT NOT NULL,
    international_sales INT NOT NULL
);

Exercﾃｭcio 1: Utilizando o INNER JOIN, encontre as vendas nacionais (domestic_sales) e internacionais (international_sales) de cada filme.
SOLUﾃﾃグ:
SELECT m.title, b.domestic_sales, b.international_sales
FROM pixar.movies m
JOIN pixar.box_office b
ON m.id = b.movie_id;

噫 Exercﾃｭcio 2: Utilizando o INNER JOIN, faﾃｧa uma busca que retorne o nﾃｺmero de vendas para cada filme que possui um nﾃｺmero maior de vendas internacionais (international_sales) do que vendas nacionais (domestic_sales).
SOLUﾃﾃグ:
SELECT m.title, (b.domestic_sales, b.international_sales) AS sales
FROM pixar.movies m
JOIN pixar.box_office b
ON m.id = b.movie_id
WHERE b.international_sales > b.domestic_sales;

噫 Exercﾃｭcio 3: Utilizando o INNER JOIN, faﾃｧa uma busca que retorne os filmes e sua avaliaﾃｧﾃ｣o (rating) em ordem decrescente.
SOLUﾃﾃグ:
SELECT m.title, b.rating
FROM pixar.movies m
JOIN pixar.box_office b
ON m.id = b.movie_id
ORDER BY b.rating DESC;

Exercﾃｭcio 4: Utilizando o LEFT JOIN, faﾃｧa uma busca que retorne todos os dados dos cinemas, mesmo os que nﾃ｣o possuem filmes em cartaz e, adicionalmente, os dados dos filmes que estﾃ｣o em cartaz nestes cinemas. Retorne os nomes dos cinemas em ordem alfabﾃｩtica.
SOLUﾃﾃグ:
SELECT t.*, m.*
FROM pixar.theater t
LEFT JOIN pixar.movies m
ON t.id = m.theater_id
ORDER BY t.name;

Exercﾃｭcio 5: Utilizando o RIGHT JOIN, faﾃｧa uma busca que retorne todos os dados dos filmes, mesmo os que nﾃ｣o estﾃ｣o em cartaz e, adicionalmente, os dados dos cinemas que possuem estes filmes em cartaz. Retorne os nomes dos cinemas em ordem alfabﾃｩtica.
SOLUﾃﾃグ:
SELECT t.*, m.*
FROM pixar.theater t
RIGHT JOIN pixar.movies m
ON t.id = m.theater_id
ORDER BY t.name;

噫 Exercﾃｭcio 6: Utilizando o INNER JOIN, selecione todas as informaﾃｧﾃｵes dos filmes que estﾃ｣o em cartaz (possuem theater_id diferente de NULL) com avaliaﾃｧﾃ｣o maior que 8.
SOLUﾃﾃグ:
SELECT m.*, b.rating
FROM pixar.movies m
JOIN pixar.box_office b
ON m.id = b.movie_id
WHERE m.theater_id IS NOT NULL
AND b.rating > 8;
